/******************************************************** * 2026 Team Up 워크숍 스케줄러 — Google Apps Script 백엔드 * * [설치 순서] * 1) 새 Google 스프레드시트 생성 * 2) 메뉴: 확장 프로그램 > Apps Script → 기존 코드를 지우고 이 코드 전체 붙여넣기 * 3) 아래 ADMIN_PASSWORD를 원하는 관리자 비밀번호로 변경 * 4) 상단 함수 선택에서 "setup" 선택 → ▶실행 (최초 1회, 권한 승인 필요) * → 시트에 7~12월 기본 일정 60개가 생성됩니다 * 5) 우측 상단 [배포] > [새 배포] > 유형: 웹 앱 * - 설명: teamup (자유) * - 다음 사용자 인증 정보로 실행: "나" * - 액세스 권한: "모든 사용자" ← 중요! * → [배포] 클릭 → 발급된 "웹 앱 URL"(https://script.google.com/macros/s/.../exec) 복사 * 6) index.html 파일 상단의 API_URL에 그 URL을 붙여넣기 * * [코드 수정 후 재배포 시] * 배포 > 배포 관리 > 연필 아이콘 > 버전: "새 버전" > 배포 * (URL은 그대로 유지됩니다) ********************************************************/ const ADMIN_PASSWORD = 'hrd@2500'; // ★ 관리자 비밀번호 — 반드시 변경하세요 const TZ = 'Asia/Seoul'; const COURSES = ['MBTI', 'Trust Building', '미션비전', '일하는 방식', 'R&R', 'Team Melting']; const SLOT_HEADER = ['id', 'courseName', 'date', 'time', 'isBooked', 'bookedWorkshopId']; const WS_HEADER = ['id', 'slotId', 'companyName', 'departmentName', 'applicantName', 'phone', 'headcount', 'note', 'password', 'createdAt']; /* ───────── 최초 1회 실행: 시트 생성 + 기본 일정 60개 ───────── */ function setup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); ss.setSpreadsheetTimeZone(TZ); // slots 시트 let slotSheet = ss.getSheetByName('slots') || ss.insertSheet('slots'); slotSheet.clearContents(); slotSheet.getRange(1, 1, 1, SLOT_HEADER.length).setValues([SLOT_HEADER]); const rows = []; const perMonth = [2, 2, 2, 2, 1, 1]; // 7월~12월 (과목당 총 10회) COURSES.forEach(function (course, ci) { [7, 8, 9, 10, 11, 12].forEach(function (month, mi) { for (let k = 0; k < perMonth[mi]; k++) { const baseDay = 3 + ci * 2 + k * 13; const d = new Date(2026, month - 1, baseDay); if (d.getDay() === 0) d.setDate(d.getDate() + 1); // 일 → 월 if (d.getDay() === 6) d.setDate(d.getDate() + 2); // 토 → 월 rows.push([ 'slot-' + ci + '-' + month + '-' + k, course, Utilities.formatDate(d, TZ, 'yyyy-MM-dd'), '10:00 ~ 16:00', false, '', ]); } }); }); slotSheet.getRange(2, 1, rows.length, SLOT_HEADER.length).setValues(rows); // workshops 시트 let wsSheet = ss.getSheetByName('workshops') || ss.insertSheet('workshops'); wsSheet.clearContents(); wsSheet.getRange(1, 1, 1, WS_HEADER.length).setValues([WS_HEADER]); } /* ───────── 공통 유틸 ───────── */ function json_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function readObjects_(name, header) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name); if (!sh) return []; const values = sh.getDataRange().getValues(); return values.slice(1).filter(function (r) { return r[0] !== ''; }).map(function (r) { const obj = {}; header.forEach(function (h, i) { let v = r[i]; if (v instanceof Date) v = Utilities.formatDate(v, TZ, 'yyyy-MM-dd'); obj[h] = v; }); return obj; }); } function writeObjects_(name, header, objects) { const sh = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name); sh.clearContents(); sh.getRange(1, 1, 1, header.length).setValues([header]); if (objects.length > 0) { const rows = objects.map(function (o) { return header.map(function (h) { const v = o[h]; return v === null || v === undefined ? '' : v; }); }); sh.getRange(2, 1, rows.length, header.length).setValues(rows); } } function isTrue_(v) { return v === true || String(v).toUpperCase() === 'TRUE'; } /* 전체 데이터 조회 — 비밀번호는 절대 클라이언트로 보내지 않음 */ function getData_() { const slots = readObjects_('slots', SLOT_HEADER).map(function (s) { return { id: s.id, courseName: s.courseName, date: s.date, time: s.time, isBooked: isTrue_(s.isBooked), bookedWorkshopId: s.bookedWorkshopId || null, }; }); const workshops = readObjects_('workshops', WS_HEADER).map(function (w) { return { id: w.id, slotId: w.slotId, companyName: w.companyName, departmentName: w.departmentName, applicantName: w.applicantName, phone: w.phone, headcount: w.headcount, note: w.note, }; }); return { ok: true, slots: slots, workshops: workshops }; } /* ───────── HTTP 핸들러 ───────── */ function doGet(e) { return json_(getData_()); } function doPost(e) { const lock = LockService.getScriptLock(); lock.waitLock(15000); // 동시 신청 충돌 방지 try { const p = JSON.parse(e.postData.contents); switch (p.action) { case 'book': return json_(book_(p)); case 'cancel': return json_(cancel_(p)); case 'adminCheck': return json_({ ok: String(p.adminPassword) === String(ADMIN_PASSWORD) }); case 'addSlot': return json_(adminGuard_(p, addSlot_)); case 'editSlot': return json_(adminGuard_(p, editSlot_)); case 'deleteSlot': return json_(adminGuard_(p, deleteSlot_)); case 'adminCancel': return json_(adminGuard_(p, adminCancel_)); case 'reset': return json_(adminGuard_(p, reset_)); default: return json_({ ok: false, error: '알 수 없는 요청입니다.' }); } } catch (err) { return json_({ ok: false, error: '서버 오류: ' + String(err) }); } finally { lock.releaseLock(); } } function adminGuard_(p, fn) { if (String(p.adminPassword) !== String(ADMIN_PASSWORD)) { return { ok: false, error: '관리자 인증에 실패했습니다.' }; } return fn(p); } /* ───────── 액션: 신청 ───────── */ function book_(p) { const slots = readObjects_('slots', SLOT_HEADER); const target = slots.find(function (s) { return s.id === p.slotId; }); if (!target) return { ok: false, error: '일정을 찾을 수 없습니다. 새로고침 후 다시 시도해주세요.' }; if (isTrue_(target.isBooked)) return { ok: false, error: '방금 다른 팀이 먼저 신청한 일정입니다. 다른 일정을 선택해주세요.' }; if (!p.companyName || !p.departmentName || !p.applicantName || !p.phone || !p.password) { return { ok: false, error: '필수 항목이 누락되었습니다.' }; } const wsId = 'ws-' + Date.now(); const workshops = readObjects_('workshops', WS_HEADER); workshops.push({ id: wsId, slotId: p.slotId, companyName: String(p.companyName), departmentName: String(p.departmentName), applicantName: String(p.applicantName), phone: String(p.phone), headcount: p.headcount || '', note: p.note || '', password: String(p.password), createdAt: Utilities.formatDate(new Date(), TZ, 'yyyy-MM-dd HH:mm'), }); target.isBooked = true; target.bookedWorkshopId = wsId; writeObjects_('workshops', WS_HEADER, workshops); writeObjects_('slots', SLOT_HEADER, slots); const data = getData_(); data.workshopId = wsId; return data; } /* ───────── 액션: 신청 취소 (신청 비밀번호 검증) ───────── */ function cancel_(p) { const workshops = readObjects_('workshops', WS_HEADER); const target = workshops.find(function (w) { return w.id === p.workshopId; }); if (!target) return { ok: false, error: '신청 내역을 찾을 수 없습니다.' }; if (String(target.password) !== String(p.password)) { return { ok: false, error: '비밀번호가 올바르지 않습니다. 분실 시 운영 담당자에게 문의해주세요.' }; } return removeWorkshop_(p.workshopId, workshops); } /* ───────── 액션: 관리자 신청 취소 (비밀번호 불필요) ───────── */ function adminCancel_(p) { return removeWorkshop_(p.workshopId, readObjects_('workshops', WS_HEADER)); } function removeWorkshop_(workshopId, workshops) { const remaining = workshops.filter(function (w) { return w.id !== workshopId; }); const slots = readObjects_('slots', SLOT_HEADER).map(function (s) { if (s.bookedWorkshopId === workshopId) { s.isBooked = false; s.bookedWorkshopId = ''; } return s; }); writeObjects_('workshops', WS_HEADER, remaining); writeObjects_('slots', SLOT_HEADER, slots); return getData_(); } /* ───────── 액션: 일정 추가 ───────── */ function addSlot_(p) { if (!p.courseName || !p.date || !p.time) return { ok: false, error: '과목, 날짜, 시간을 모두 입력해주세요.' }; const slots = readObjects_('slots', SLOT_HEADER); slots.push({ id: 'slot-custom-' + Date.now(), courseName: String(p.courseName), date: String(p.date), time: String(p.time), isBooked: false, bookedWorkshopId: '', }); writeObjects_('slots', SLOT_HEADER, slots); return getData_(); } /* ───────── 액션: 일정 수정 ───────── */ function editSlot_(p) { const slots = readObjects_('slots', SLOT_HEADER); const target = slots.find(function (s) { return s.id === p.slotId; }); if (!target) return { ok: false, error: '일정을 찾을 수 없습니다.' }; target.courseName = String(p.courseName); target.date = String(p.date); target.time = String(p.time); writeObjects_('slots', SLOT_HEADER, slots); return getData_(); } /* ───────── 액션: 일정 삭제 (연결된 신청도 함께 삭제) ───────── */ function deleteSlot_(p) { const slots = readObjects_('slots', SLOT_HEADER); const remaining = slots.filter(function (s) { return s.id !== p.slotId; }); const workshops = readObjects_('workshops', WS_HEADER).filter(function (w) { return w.slotId !== p.slotId; }); writeObjects_('slots', SLOT_HEADER, remaining); writeObjects_('workshops', WS_HEADER, workshops); return getData_(); } /* ───────── 액션: 데이터 초기화 ───────── */ function reset_() { setup(); return getData_(); }
